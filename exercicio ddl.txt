CREATE DATABASE vendas
USE vendas

CREATE TABLE produto(
id int AUTO_INCREMENT,
nome varchar not null (20),
qtd_produto integer,
descrição varchar(20) NOT null UNIQUE,
precoproduto float,
PRIMARY KEY(id)
);

CREATE TABLE notafiscal(
id int AUTO_INCREMENT,
dataSaida_NF timestamp,
dataEntrada_NF timestamp,
valornf float,
PRIMARY KEY(id)
);

create table itens(
cod_itens int not unique,
qtd_itens int,
idProduto int,
idNota_fiscal int,
primary key(idProduto, idNota-fiscal),
foreign key (IdProduto) references produto(id),
foreign key (idNota_fiscal) references Nota_fiscal(id)
);
